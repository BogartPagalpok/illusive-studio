import { useEffect } from 'react'; export function useHoveringPenFavicon() { useEffect(() => { const canvas = document.createElement('canvas'); // 64x64 ensures it remains sharp on high-DPI retina displays canvas.width = 64; canvas.height = 64; const ctx = canvas.getContext('2d'); let link = document.querySelector("link[rel*='icon']") as HTMLLinkElement; if (!link) { link = document.createElement('link'); document.head.appendChild(link); } link.type = 'image/png'; link.rel = 'icon'; let animationId: number; const animate = () => { if (!ctx) return; // Dynamically pull the active theme's accent color const rootStyle = getComputedStyle(document.documentElement); const accent = rootStyle.getPropertyValue('--accent').trim() || '#00ffcc'; // Use a secondary contrasting color for the bezier nodes (defaulting to purple) const nodeColor = '#a855f7'; ctx.clearRect(0, 0, 64, 64); // --- 1. Draw Bezier Curve & Nodes (Static Background) --- ctx.beginPath(); ctx.moveTo(12, 44); ctx.bezierCurveTo(12, 20, 52, 20, 52, 44); ctx.strokeStyle = nodeColor; ctx.lineWidth = 3; ctx.stroke(); // Square Nodes ctx.fillStyle = nodeColor; ctx.fillRect(8, 40, 8, 8); // Left ctx.fillRect(48, 40, 8, 8); // Right ctx.fillRect(28, 16, 8, 8); // Top Center // --- 2. Calculate Smooth Hover Offset --- // Divisor controls speed; multiplier controls height const time = Date.now() / 250; const yOffset = Math.sin(time) * 4; // --- 3. Draw Pen Tool (Hovering Element) --- ctx.save(); ctx.translate(0, yOffset); // Outer Pen Shape ctx.beginPath(); ctx.moveTo(32, 24); // Sharp Tip ctx.lineTo(44, 46); // Right Edge ctx.lineTo(38, 46); // Right Inner Base ctx.lineTo(38, 56); // Right Bottom ctx.lineTo(26, 56); // Left Bottom ctx.lineTo(26, 46); // Left Inner Base ctx.lineTo(20, 46); // Left Edge ctx.closePath(); ctx.strokeStyle = accent; ctx.lineWidth = 4; ctx.lineJoin = 'round'; ctx.stroke(); // Center Vertical Line ctx.beginPath(); ctx.moveTo(32, 24); ctx.lineTo(32, 46); ctx.stroke(); // Bottom Horizontal Detail Line ctx.beginPath(); ctx.moveTo(26, 51); ctx.lineTo(38, 51); ctx.stroke(); ctx.restore(); // Apply the generated frame to the browser tab link.href = canvas.toDataURL('image/png'); // Loop the animation animationId = requestAnimationFrame(animate); }; animate(); return () => cancelAnimationFrame(animationId); }, []); }
